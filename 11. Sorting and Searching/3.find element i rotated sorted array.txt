arr=[15,16,19,20,25,1,3,4,5,7,10,14]

def binsearch(arr,val,i,j):
	if i>j:
		return None
	mid=(i+j)/2
	if arr[mid]==val:
		return mid
	if arr[i]<=val and arr[i]>arr[mid] or arr[i]<arr[mid] and arr[i]<=val<arr[mid]:
		return binsearch(arr,val,i,mid-1)
	else:
		return binsearch(arr,val,mid+1,j)


val=19
print binsearch(arr,val,0,len(arr))
